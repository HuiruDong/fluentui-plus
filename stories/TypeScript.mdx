# TypeScript 支持

Fluent UI Plus 使用 TypeScript 开发，提供完整的类型定义支持。

## 📖 类型定义

所有组件都包含完整的 TypeScript 类型定义，无需额外安装 `@types` 包。

### 组件 Props 类型

```tsx
import React from 'react';
import { Button, ButtonProps, Tag, TagProps } from 'fluentui-plus';

// 使用组件 Props 类型
const CustomButton: React.FC<ButtonProps> = (props) => {
  return <Button {...props} />;
};

// 扩展组件 Props
interface ExtendedButtonProps extends ButtonProps {
  customProp?: string;
}

const ExtendedButton: React.FC<ExtendedButtonProps> = ({
  customProp,
  ...buttonProps
}) => {
  return (
    <div>
      {customProp && <span>{customProp}</span>}
      <Button {...buttonProps} />
    </div>
  );
};
```

### 组件 Ref 类型

```tsx
import React, { useRef } from 'react';
import { Button } from 'fluentui-plus';

function App() {
  const buttonRef = useRef<HTMLButtonElement>(null);

  const handleClick = () => {
    if (buttonRef.current) {
      buttonRef.current.focus();
    }
  };

  return (
    <div>
      <Button ref={buttonRef} onClick={handleClick}>
        点击我
      </Button>
    </div>
  );
}
```

## 🎯 常用类型

### Button 组件类型

```tsx
interface ButtonProps extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  variant?: 'primary' | 'default' | 'secondary' | 'outline' | 'text' | 'link';
  size?: 'small' | 'medium' | 'large';
  block?: boolean;
  loading?: boolean;
  icon?: React.ReactNode;
  iconPosition?: 'start' | 'end';
  disabled?: boolean;
  children?: React.ReactNode;
}
```

### Tag 组件类型

```tsx
interface TagProps {
  closeIcon?: boolean | React.ReactNode;
  color?: string;
  bordered?: boolean;
  className?: string;
  style?: React.CSSProperties;
  onClose?: (e: React.MouseEvent<HTMLElement, MouseEvent>) => void;
  onClick?: (e: React.MouseEvent<HTMLElement, MouseEvent>) => void;
  children?: React.ReactNode;
}

interface CheckableTagProps {
  checked?: boolean;
  onChange?: (checked: boolean) => void;
  className?: string;
  style?: React.CSSProperties;
  children?: React.ReactNode;
}
```

## 📚 高级用法

### 泛型组件

```tsx
interface SelectOption<T = any> {
  label: string;
  value: T;
}

interface SelectProps<T = any> {
  options: SelectOption<T>[];
  value?: T;
  onChange?: (value: T) => void;
}

// 使用泛型
const stringOptions: SelectOption<string>[] = [
  { label: '选项1', value: 'option1' },
  { label: '选项2', value: 'option2' },
];

const numberOptions: SelectOption<number>[] = [
  { label: '选项1', value: 1 },
  { label: '选项2', value: 2 },
];
```

### 条件类型

```tsx
type ButtonVariant = 'primary' | 'default' | 'secondary';

interface ConditionalButtonProps {
  variant: ButtonVariant;
  // 根据 variant 条件性地要求不同的 props
  customProp: ButtonVariant extends 'primary' ? string : never;
}
```
